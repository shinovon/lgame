Инструкция для оформления update.json клиента
Версия 2 от 22.07.2021
{
	id: (string) иденификатор клиента
	date: (string) дата последнего изменения, формат: ДД.ММ.ГГГГ (пример: 22.07.2021)
	format: (int) 1 (версия формата данных)
	update_build: (int) номер обновления понимаемая лаунчером
	asset_index: (string) индекс ассетов
	integrity_check: (object) конфигурация для проверки целостности
	{
		dirs: (array) обязательные папки сборки, формат записей: {"type": "тип проверки", "path": "название папки/файла"}
		[
			{ (описание параметров записи)
				type: (string) тип проверки ("exists" - существует ли папка/файл)
				path: (string) название папки/файла
			}
		],
		natives: (object) раздел для проверки целостности папки с нативными библиотеками
		{
			checksum_algorithm: (string) алгорити хешей (рекомендованно - SHA-1)
			checksums: (object) хеши модов, формат записей: "файл": "хеш"
			{ (пример записей)
				"1.dll": "hash",
				"2.dll": "hash"
			}
		},
		libraries: (object) раздел для проверки целостности библиотек
		{
			checksum_algorithm: (string) алгорити хешей (рекомендованно - SHA-1)
			checksums: (object) хеши модов, формат записей: "файл": "хеш"
			{ (пример записей)
				"1.jar": "hash",
				"2.jar": "hash"
			}
		}
	},
	update: (object) конфигурация для обновления
	{
		updates_check: (object) проверка наличия обновлений
		{
			type: (string) тип проверки наличия обновлений, на данный момент только один: update_build
		},
		update_scripts: (object) сценарии
		{
			install_script: (object) сценарий первого скачивания
			{
				download: (array)
				unzip: (array)
			},
			update_script: (object) сценарий обновления
			{
				remove: (array)
				download: (array)
				unzip: (array)
			}
		}
	},
	run: (object) конфигуриция для старта клиента
	{
		mainclass: (string) запускной класс клиента
		tweak_classes: (array) массив строк с классами твикерами
	}
}
